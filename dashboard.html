export default function handler(req, res) { const cookies = req.headers.cookie || ''; const match = cookies.match(/tncity_user=([^;]+)/); if (!match) { return res.status(401).json({ error: 'غير مسجل الدخول' }); } try { const user = JSON.parse(Buffer.from(match[1], 'base64').toString('utf-8')); res.json(user); } catch(e) { res.status(401).json({ error: 'بيانات غير صالحة' }); } }
